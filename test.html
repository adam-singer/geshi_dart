#import("dart:io");
class Printer extends Isolate {
  main() {
    port.receive((message, replyTo) {
      if (message == null) port.close();
      else print(message);
    });
  }
}
 
main() {
  var v = "10" + 10.toString() + "${v}";
  new Printer().spawn().then((port) {
    for (var message in ["Hello", "from", "other", "isolate"]) {
      port.send(message);  
    }
    port.send(null);
  });
}
